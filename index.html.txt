    <script>
        // Инициализация Telegram Mini App
        const tg = window.Telegram.WebApp;
        tg.expand();

        // Базовый словарь (можешь добавлять сюда новые слова)
        const words = [
            { w: "Słońce", t: "солнце", r: "to" },
            { w: "Kobieta", t: "женщина", r: "ta" },
            { w: "Mężczyzna", t: "мужчина", r: "ten" },
            { w: "Okno", t: "окно", r: "to" },
            { w: "Noc", t: "ночь", r: "ta" },
            { w: "Dzień", t: "день", r: "ten" },
            { w: "Naczynie", t: "посуда", r: "to" },
            { w: "Rzecz", t: "вещь", r: "ta" },
            { w: "Samochód", t: "машина", r: "ten" },
            { w: "Książka", t: "книга", r: "ta" },
            { w: "Dziecko", t: "ребенок", r: "to" },
            { w: "Miłość", t: "любовь", r: "ta" }
        ];

        let currentWord = {};
        let score = 0;
        let total = 0;

        function nextWord() {
            const card = document.getElementById('card');
            // Сбрасываем классы правильного/неправильного ответа
            card.className = 'word-card';
            
            // Выбираем случайное слово
            currentWord = words[Math.floor(Math.random() * words.length)];
            document.getElementById('word').innerText = currentWord.w;
            document.getElementById('translation').innerText = currentWord.t;
        }

        function check(gender) {
            total++;
            const card = document.getElementById('card');
            
            if (gender === currentWord.r) {
                score++;
                card.classList.add('correct');
                if (tg.HapticFeedback) tg.HapticFeedback.notificationOccurred('success');
            } else {
                card.classList.add('wrong');
                if (tg.HapticFeedback) tg.HapticFeedback.notificationOccurred('error');
            }

            // Обновляем статистику (Wynik - это "счет/результат" по-польски)
            document.getElementById('stats').innerText = `WYNIK: ${score} / ${total}`;
            
            // Задержка перед следующим словом, чтобы было видно результат
            setTimeout(nextWord, 500);
        }

        // Запускаем первое слово при загрузке
        nextWord();
    </script>
